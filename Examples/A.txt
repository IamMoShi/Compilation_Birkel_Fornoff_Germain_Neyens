with Ada.Text_IO; use Ada.Text_IO;

procedure Fibonacci is
    N, C  : Integer := 0;
    A : Integer;
    B : Integer;
    Next : Integer;

begin
    A := 0;
    B := 1;

    Get(N);

    for I in 1..N loop
        if I <= 1 then
            Next := I;
        else
            Next := A + B;
            A := B;
            B := Next;
        end if;
    end loop;

end Fibonacci;
